def find_path(grid, start_pos, end_pos, width, height, diagonal=False):
    import heapq
    import math

    def heuristic(y1, x1, y2, x2):
        if diagonal:
            return math.sqrt((y1-y2)**2 + (x1-x2)**2)
        else:
            return abs(y1-y2) + abs(x1-x2)

    def get_neighbors(y, x):
        neighbors = []
        directions = [(-1,0), (1,0), (0,-1), (0,1)]
        if diagonal:
            directions.extend([(-1,-1), (-1,1), (1,-1), (1,1)])

        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width and grid[ny][nx] != 1:
                neighbors.append((ny, nx))
        return neighbors

    start_y, start_x = start_pos
    end_y, end_x = end_pos

    # ä¼˜å…ˆé˜Ÿåˆ—ï¼š(f_score, g_score, y, x, parent)
    heap = [(heuristic(start_y, start_x, end_y, end_x), 0, start_y, start_x, None)]

    # è®°å½•è®¿é—®ä¿¡æ¯
    visited = []
    visited_set = set()

    # è®°å½•æ¯ä¸ªä½ç½®çš„çˆ¶èŠ‚ç‚¹å’Œgå€¼
    parent = {}
    g_score = {start_pos: 0}
    f_score = {start_pos: heuristic(start_y, start_x, end_y, end_x)}

    while heap:
        current_f, current_g, y, x, current_parent = heapq.heappop(heap)
        current_pos = (y, x)

        if current_pos in visited_set:
            continue

        visited_set.add(current_pos)
        visited.append([y, x])

        if current_pos == end_pos:
            # æ‰¾åˆ°ç»ˆç‚¹ï¼Œé‡å»ºè·¯å¾„
            path = []
            while current_pos in parent:
                path.append(list(current_pos))
                current_pos = parent[current_pos]
            path.append(list(start_pos))
            path.reverse()

            return {
                'visited': visited,
                'path': path,
                'found': True
            }

        for ny, nx in get_neighbors(y, x):
            neighbor_pos = (ny, nx)
            tentative_g = current_g + (1.414 if diagonal else 1)

            if neighbor_pos not in g_score or tentative_g < g_score[neighbor_pos]:
                parent[neighbor_pos] = current_pos
                g_score[neighbor_pos] = tentative_g
                f_score[neighbor_pos] = tentative_g + heuristic(ny, nx, end_y, end_x)
                heapq.heappush(heap, (f_score[neighbor_pos], tentative_g, ny, nx, current_pos))

    return {
        'visited': visited,
        'path': [],
        'found': False
    }

def find_path(grid, start_pos, end_pos, width, height, diagonal=False):
    from collections import deque

    def get_neighbors(y, x):
        neighbors = []
        directions = [(-1,0), (1,0), (0,-1), (0,1)]
        if diagonal:
            directions.extend([(-1,-1), (-1,1), (1,-1), (1,1)])

        for dy, dx in directions:
            ny, nx = y + dy, x + dx
            if 0 <= ny < height and 0 <= nx < width and grid[ny][nx] != 1:
                neighbors.append((ny, nx))
        return neighbors

    start_y, start_x = start_pos
    end_y, end_x = end_pos

    # é˜Ÿåˆ—ï¼š(y, x, parent)
    queue = deque([(start_y, start_x, None)])

    # è®°å½•è®¿é—®ä¿¡æ¯
    visited = []
    visited_set = set([(start_y, start_x)])

    # è®°å½•çˆ¶èŠ‚ç‚¹
    parent = {}

    while queue:
        y, x, current_parent = queue.popleft()
        current_pos = (y, x)

        visited.append([y, x])

        if current_pos == end_pos:
            # æ‰¾åˆ°ç»ˆç‚¹ï¼Œé‡å»ºè·¯å¾„
            path = []
            while current_pos in parent:
                path.append(list(current_pos))
                current_pos = parent[current_pos]
            path.append(list(start_pos))
            path.reverse()

            return {
                'visited': visited,
                'path': path,
                'found': True
            }

        for ny, nx in get_neighbors(y, x):
            neighbor_pos = (ny, nx)
            if neighbor_pos not in visited_set:
                visited_set.add(neighbor_pos)
                parent[neighbor_pos] = current_pos
                queue.append((ny, nx, current_pos))

    return {
        'visited': visited,
        'path': [],
        'found': False
    }
